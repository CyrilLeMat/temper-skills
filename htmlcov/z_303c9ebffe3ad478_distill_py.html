65def _parse_feature_spec(spec: str) -> tuple[str, dict]: 

66 """Parse a schema_critic 'name: type — why' into (name, json-schema-fragment). 

67 

68 Best-effort: unknown/compound types (Literal[...], a|b) fall back to string. Returns 

69 ("", {}) if the spec has no parseable name.""" 

70 head = spec.split("—", 1)[0].split(" - ", 1)[0].strip() 

71 name, _, type_part = head.partition(":") 

72 name = name.strip() 

73 if not name or not name.isidentifier(): 

74 return "", {} 

75 t = type_part.strip().lower() 

76 frag = {"type": _JSON_TYPE.get(t, "string")} 

77 return name, frag

80def _feature_used_in_tree(name: str, tree: ProposedTree) -> bool: 

81 """Does any node condition branch on this feature (word-boundary match)?""" 

82 pat = re.compile(rf"\b{re.escape(name)}\b") 

83 return any(pat.search(n.condition or "") for n in tree.nodes)

113def _render_tree(tree: ProposedTree) -> str: 

114 lines = [] 

115 for i, n in enumerate(tree.nodes, 1): 

116 gz = f" [gray_zone: {n.gray_zone}]" if n.gray_zone else "" 

117 lines.append(f" n{i}: if ({n.condition}) -> {n.outcome}{gz}") 

118 lines.append(f" default -> {tree.default_outcome}") 

119 return "\n".join(lines)

122def _sources_block(sources: Sources) -> str: 

123 parts = [] 

124 if sources.skill_text: 

125 parts.append( 

126 "ORIGINAL SKILL (the prompt logic being migrated — extract its routing " 

127 f"logic, not its prose generation):\n{sources.skill_text}\n" 

128 ) 

129 parts += [ 

130 "SCHEMA (pre-computed structured features the tree may branch on):", 

131 json.dumps(sources.json_schema, indent=2), 

132 ] 

133 if sources.constraints: 

134 parts.append("\nCONSTRAINTS (hard ones are non-negotiable):") 

135 for i, c in enumerate(sources.constraints, 1): 

136 tag = " [HARD]" if c.get("hard") else "" 

137 parts.append(f" constraints#{i}{tag}: {c['rule']}") 

138 if sources.examples: 

139 parts.append("\nEXAMPLES (ratified cases — waypoints, not a dataset):") 

140 for i, e in enumerate(sources.examples, 1): 

141 parts.append(f" examples#{i}: input={e.get('input')} -> expected={e.get('expected')}") 

142 return "\n".join(parts)

197def _critique(backend: Backend, sources: Sources, persona: Persona, tree: ProposedTree, 

198 closed: list[str] | None = None) -> PersonaVerdict: 

199 if persona.name == SCHEMA_CRITIC.name: 

200 tests_clause = ( 

201 "Leave `proposed_tests` EMPTY. Your job is the schema's EXPRESSIVENESS: judge " 

202 "whether the source implies a feature the schema cannot express (so the tree is " 

203 "forced to punt or over-approximate). If so, set verdict='schema_too_thin' and " 

204 "list each gap in `proposed_features` as 'name: type — why the source needs it'. " 

205 "If the schema can express everything the source decides on, score high and say so." 

206 ) 

207 elif persona.name == OUTCOME_CRITIC.name: 

208 tests_clause = ( 

209 "Leave `proposed_tests` EMPTY. Your job is the OUTCOME set's expressiveness: judge " 

210 "whether the source implies a distinct answer the current outcomes cannot express, " 

211 "so two genuinely different correct answers are forced to collapse into one label " 

212 "(or a case is routed to an outcome that is only approximately right). If so, set " 

213 "verdict='outcome_too_coarse' and list each missing outcome in `proposed_outcomes` " 

214 "as 'outcome — why (which two cases collapse today)'. If the outcome set can express " 

215 "every answer the source calls for, score high and say so." 

216 ) 

217 elif persona.name == OVERENGINEERING_CRITIC.name: 

218 tests_clause = ( 

219 "Leave `proposed_tests` EMPTY — your job is to remove unjustified complexity, " 

220 "not to add test cases." 

221 ) 

222 else: 

223 tests_clause = ( 

224 "Whenever you find a flaw, ALSO populate `proposed_tests`: one or more concrete " 

225 "cases, each a full `input` (a feature assignment over the schema features), the " 

226 "`expected` outcome you believe is correct, and a one-line `rationale`. These " 

227 "extend a human-ratified validation set — give your best-judgment label; a human " 

228 "ratifies it, it is not trusted blindly. Use exact schema feature names and values." 

229 ) 

230 # Settled trade-offs the proposer has already ruled on. Without this, a persona re-raises 

231 # the same rejected point every round and pins its own score low forever — so the panel 

232 # never climbs even on a tree that's as good as it will get. 

233 closed_clause = "" 

234 if closed: 

235 closed_clause = ( 

236 "\n\nSETTLED — the proposer has deliberately accepted these as gray zones or " 

237 "rejected a prior critique about them. Do NOT re-raise them or lower your score " 

238 "for them; they are decided. Score only on NEW issues:\n" 

239 + "\n".join(f" - {c}" for c in closed) 

240 ) 

241 user = ( 

242 f"Your angle ({persona.name}): {persona.style}\n\n" 

243 f"{_sources_block(sources)}\n\n" 

244 f"CURRENT TREE:\n{_render_tree(tree)}\n\n" 

245 f"Review the tree strictly from your angle. Set persona to '{persona.name}'.\n\n" 

246 f"{tests_clause}{closed_clause}" 

247 ) 

248 return backend.complete(PERSONA_SYSTEM, user, PersonaVerdict)

251def _critique_all( 

252 backend: Backend, sources: Sources, personas: list[Persona], tree: ProposedTree, 

253 closed: list[str] | None = None, 

254) -> list[PersonaVerdict]: 

255 """Run every persona's critique of the SAME tree concurrently (they're independent). 

256 

257 Order-preserving (executor.map), so the panel and tests stay deterministic. This is 

258 the bulk of a round, so parallelizing it cuts round latency roughly N×. 

259 """ 

260 if len(personas) <= 1: 

261 return [_critique(backend, sources, p, tree, closed) for p in personas] 

262 with ThreadPoolExecutor(max_workers=len(personas)) as ex: 

263 return list(ex.map(lambda p: _critique(backend, sources, p, tree, closed), personas))

266def _arbitrate( 

267 backend: Backend, sources: Sources, tree: ProposedTree, verdicts: list[PersonaVerdict], 

268 incremental: bool = False, regressed_last: bool = False, 

269) -> ProposerArbitration: 

270 crit = "\n".join( 

271 f" {v.persona} (score {v.score}, {v.verdict}): {v.detail}" 

272 + (f" | case: {v.proposed_case}" if v.proposed_case else "") 

273 for v in verdicts 

274 ) 

275 incr = ( 

276 "INCREMENTAL RUN: the CURRENT TREE is an existing, previously-converged tree; a " 

277 "change to constraints/sources prompted this run. Preserve every node the change " 

278 "does NOT affect — keep its condition and outcome verbatim. Only add, modify, or " 

279 "remove nodes the new constraints/sources (or a surviving critique) require. " 

280 "Minimize churn.\n\n" 

281 if incremental else "" 

282 ) 

283 caution = ( 

284 "NOTE: your previous revision REGRESSED (it lowered the validation pass-rate or " 

285 "broke a ratified example) and was DISCARDED — the CURRENT TREE below is the best " 

286 "version so far. Revise it only if you can RAISE the pass-rate, or remove/merge a " 

287 "node WITHOUT dropping any case. Do not collapse or reorder branches when that " 

288 "loses coverage; if no safe improvement exists, return the tree unchanged.\n\n" 

289 if regressed_last else "" 

290 ) 

291 user = ( 

292 f"{incr}{caution}" 

293 f"{_sources_block(sources)}\n\n" 

294 f"CURRENT TREE:\n{_render_tree(tree)}\n\n" 

295 f"PERSONA VERDICTS:\n{crit}\n\n" 

296 "Arbitrate. For each persona, decide kept/changed/rejected with a one-line " 

297 "rationale. Then output the improved tree. Add a branch only if a critique " 

298 "justifies it AND a domain expert would write it by hand; collapse branches the " 

299 "overengineering_critic flags as unnecessary ONLY when doing so changes no outcome. " 

300 "Respect every HARD constraint. " 

301 "Give a convergence_estimate (0-100) for how settled the tree now is." 

302 ) 

303 # The arbiter runs under its OWN system prompt, not PROPOSER_SYSTEM: the agent that 

304 # rules kept/changed/rejected must not be the one defending its own draft (§4.2). 

305 return backend.complete(ARBITER_SYSTEM, user, ProposerArbitration)

320def _sanitize(tree: ProposedTree) -> ProposedTree: 

321 """Drop nodes whose condition won't compile so a malformed branch from the model can 

322 never reach scoring or the exported file — a condition that can't compile can't run, 

323 and dropping it is strictly safer than shipping a SyntaxError.""" 

324 good = [n for n in tree.nodes if _condition_ok(n.condition)] 

325 if len(good) == len(tree.nodes): 

326 return tree 

327 return ProposedTree(nodes=good, default_outcome=tree.default_outcome)

330def distill( 

331 sources: Sources, 

332 adversaries: list[Persona] | None = None, 

333 profile: str = "standard", 

334 model: str = "claude-sonnet-4-6", 

335 backend: Backend | None = None, 

336 gate: Gate | None = None, 

337 fn_name: str = "decide", 

338 seed_tree: ProposedTree | None = None, 

339 seed_survival: dict[str, int] | None = None, 

340 propose_examples: bool = True, 

341 evolve_schema: bool = True, 

342 checkpoint: "Callable[[DecisionTree], None] | None" = None, 

343) -> DecisionTree: 

344 """Run the adversarial loop and return a deterministic DecisionTree. 

345 

346 Pass ``seed_tree`` (with ``seed_survival``) to start from an existing tree 

347 instead of a blank draft — incremental mode (see incremental.recrystallize). 

348 

349 ``propose_examples`` (on by default) builds a validation set as the loop runs: every 

350 persona except the overengineering_critic contributes discriminating cases each round. 

351 The loop SCORES the tree against these (adversary-authored) cases to pick the best 

352 tree and to decide convergence — so it returns a good final result without waiting on 

353 human ratification. Ratified examples, when supplied, rank ahead of the proposed ones 

354 and are never traded away. The proposed set is attached as ``tree.proposed_examples`` 

355 for the user to ratify and harden later; set ``propose_examples=False`` to skip it. 

356 

357 ``checkpoint`` (if given) is called with the current finalized tree after every round, 

358 so a caller can persist progress — you can follow the tree as it evolves and a late 

359 crash never throws away the rounds already paid for. 

360 """ 

361 if profile not in PROFILES: 

362 raise ValueError(f"unknown profile {profile!r}; choose from {list(PROFILES)}") 

363 max_rounds, stop_quiet, _interactive, provenance = PROFILES[profile] 

364 

365 # Co-evolving schema: mutate a private copy so the caller's Sources is never touched. 

366 if evolve_schema: 

367 sources = copy.copy(sources) 

368 sources.json_schema = copy.deepcopy(sources.json_schema) 

369 sources.json_schema.setdefault("properties", {}) 

370 

371 personas = list(adversaries) if adversaries is not None else list(PROFILE_PERSONAS[profile]) 

372 # The overengineering_critic is always on (§5.5). 

373 if not any(p.name == OVERENGINEERING_CRITIC.name for p in personas): 

374 personas.append(OVERENGINEERING_CRITIC) 

375 

376 backend = backend or auto_backend(model) 

377 model_tag = f"{backend.model} via {backend.name}" 

378 incremental = seed_tree is not None 

379 if incremental: 

380 tree = seed_tree 

381 survival = dict(seed_survival or {_node_key(n.condition): 1 for n in tree.nodes}) 

382 else: 

383 tree = _sanitize(_initial_tree(backend, sources)) 

384 survival = {_node_key(n.condition): 1 for n in tree.nodes} 

385 stale_rounds = 0 

386 regressed_last = False 

387 last_arbitration: ProposerArbitration | None = None 

388 

389 # The proposed validation set grows as the loop runs: every persona (except the 

390 # overengineering_critic) contributes discriminating cases each round, deduped here 

391 # by input and against any ratified examples. Zero extra LLM calls — these ride 

392 # along in the critiques the personas already return. 

393 proposed: dict[str, dict] = {} 

394 existing_inputs = {_canon(e["input"]) for e in sources.examples} 

395 # Advisory expressiveness findings, deduped across rounds. Outcomes are free-form in the 

396 # library (no ceiling), so the outcome_critic stays advisory here. Schema features co-evolve: 

397 # watching[name] = (round_added, spec) — added, on probation until it earns a branch 

398 # earned[name] = spec — branched on; now a permanent schema member 

399 # schema_gaps = specs proposed but reverted (earned no branch in _EARN_ROUNDS) 

400 schema_gaps: dict[str, None] = {} 

401 outcome_gaps: dict[str, None] = {} 

402 watching: dict[str, tuple[int, str]] = {} 

403 earned: dict[str, str] = {} 

404 

405 # We SCORE the tree against those proposed cases, not just the ratified ones: the 

406 # point of the tool is to hand back a good final tree without waiting on a human to 

407 # ratify every step. The labels are written by the ADVERSARIAL personas, not the 

408 # proposer, so this is "satisfy your critics", not "grade your own homework". 

409 # Ranking is lexicographic — (ratified pass-rate, proposed pass-rate, persona score) — 

410 # so ratified ground truth, when present, always wins and is never traded away to 

411 # match a proposed label. `candidates` keeps each round's tree; the winner is chosen 

412 # by re-scoring them all against the FINAL proposed set (fair: one denominator). 

413 # `stop_key` is the running plateau tracker. 

414 candidates: list[tuple[ProposedTree, ProposerArbitration | None, float]] = [(tree, None, -1.0)] 

415 stop_key = (_agreement(tree, sources, fn_name) or 1.0, -1.0, -1.0) 

416 

417 for r in range(1, max_rounds + 1): 

418 # Feed the panel the proposer's settled trade-offs (the working tree's gray zones + 

419 # last round's rejected critiques) so a persona stops re-scoring a decided point down. 

420 closed = [n.gray_zone for n in tree.nodes if n.gray_zone] 

421 if last_arbitration: 

422 closed += [f"{e.persona}: {e.rationale}" for e in last_arbitration.entries 

423 if e.decision == "rejected"] 

424 try: 

425 verdicts = _critique_all(backend, sources, personas, tree, closed) 

426 for v in verdicts: 

427 for oc in v.proposed_outcomes: 

428 outcome_gaps.setdefault(oc, None) 

429 for feat in v.proposed_features: 

430 if evolve_schema: 

431 # Grow the schema: add the feature so THIS round's arbiter (and the 

432 # next proposer) can branch on it. It's on probation until it earns a 

433 # branch (below); if it never does, it's reverted, not shipped. 

434 name, frag = _parse_feature_spec(feat) 

435 if name and name not in sources.json_schema["properties"] \ 

436 and name not in earned: 

437 sources.json_schema["properties"][name] = frag 

438 watching.setdefault(name, (r, feat)) 

439 else: 

440 schema_gaps.setdefault(feat, None) # advisory-only (legacy behavior) 

441 if propose_examples: 

442 _harvest_proposed(verdicts, r, proposed, existing_inputs) 

443 arbitration = _arbitrate(backend, sources, tree, verdicts, 

444 incremental=incremental, regressed_last=regressed_last) 

445 except Exception: 

446 # A transient backend failure shouldn't throw away the rounds already 

447 # paid for. With nothing to salvage (round 1), re-raise; otherwise keep 

448 # the best tree so far and finalize it. 

449 if last_arbitration is None: 

450 raise 

451 break 

452 new_tree = _sanitize(arbitration.tree) 

453 last_arbitration = arbitration 

454 

455 # Earn-a-branch-or-revert: a probationary feature that the tree now branches on is 

456 # earned (permanent); one that hasn't earned a branch within _EARN_ROUNDS is reverted 

457 # out of the schema and recorded as an advisory gap. Keeps schema size ≤ tree size. 

458 if evolve_schema and watching: 

459 for name in list(watching): 

460 added_round, spec = watching[name] 

461 if _feature_used_in_tree(name, new_tree): 

462 earned[name] = spec 

463 del watching[name] 

464 elif r - added_round >= _EARN_ROUNDS: 

465 sources.json_schema["properties"].pop(name, None) 

466 schema_gaps.setdefault(spec, None) 

467 del watching[name] 

468 

469 for key in {_node_key(n.condition) for n in new_tree.nodes}: 

470 survival[key] = survival.get(key, 0) + 1 

471 

472 # Score this round's ATTEMPT against the ratified examples (authoritative) and the 

473 # proposed cases (the panel's adversarial expectations). 

474 agreement = _agreement(new_tree, sources, fn_name) 

475 prop_items = list(proposed.values()) 

476 prop_passed = _score(new_tree, prop_items, sources.feature_names, fn_name) 

477 result = RoundResult(r, max_rounds, verdicts, arbitration, new_tree, dict(survival), 

478 agreement, proposed_count=len(prop_items), 

479 proposed_passed=prop_passed, 

480 ratified_count=len(sources.examples)) 

481 candidates.append((new_tree, arbitration, result.mean_score)) 

482 

483 # Hill-climb: adopt the attempt as the working tree ONLY if it is not worse than the 

484 # current one. A regressing round is discarded and the NEXT round re-attempts from 

485 # the best tree (with a caution to the proposer) — so one bad collapse can't poison 

486 # the rest of the run. The final export still picks the best candidate across rounds. 

487 new_q = (agreement if agreement is not None else 1.0, 

488 prop_passed / len(prop_items) if prop_items else 0.0, 

489 -len(new_tree.nodes)) 

490 if new_q >= _quality(tree, sources, prop_items, fn_name): 

491 tree = new_tree 

492 regressed_last = False 

493 else: 

494 regressed_last = True 

495 

496 if checkpoint: # persist the current best (working) tree — followable + crash-safe 

497 checkpoint(_finalize(tree, last_arbitration, survival, sources, model_tag, 

498 profile, provenance, fn_name)) 

499 

500 # Convergence: once no round beats the best for `stop_quiet` rounds, the loop has 

501 # plateaued — stop, whether the plateau is high (a good tree) or low (can't improve). 

502 round_key = (agreement if agreement is not None else 1.0, 

503 prop_passed / len(prop_items) if prop_items else 0.0, 

504 result.mean_score) 

505 if round_key > stop_key: 

506 stop_key = round_key 

507 stale_rounds = 0 

508 else: 

509 stale_rounds += 1 

510 

511 decision = gate(result) if gate else "continue" 

512 if decision == "abort": 

513 raise KeyboardInterrupt("distillation aborted by gate") 

514 if decision == "stop": 

515 break 

516 if stale_rounds >= stop_quiet: 

517 break 

518 

519 # Pick the tree to ship by re-scoring every candidate against the FINAL proposed set 

520 # (+ ratified), so a later round isn't judged on a smaller set than an earlier one. 

521 # `survival` is cumulative and keyed by condition, so it gives the right 

522 # rounds_survived for whichever tree wins. 

523 best_tree, best_arbitration = _select_best(candidates, sources, list(proposed.values()), fn_name) 

524 arb_for_final = best_arbitration if best_arbitration is not None else last_arbitration 

525 

526 # Final earn-a-branch reconciliation against the WINNING tree: a co-evolution-added feature 

527 # ships only if the shipped tree actually branches on it; the rest revert to advisory gaps. 

528 # (Original, user-supplied schema features are never reverted — they're the naive seed.) 

529 added_features: list[str] = [] 

530 if evolve_schema: 

531 for name in list(earned) + list(watching): 

532 spec = earned.get(name) or watching[name][1] 

533 if _feature_used_in_tree(name, best_tree): 

534 added_features.append(spec) 

535 else: 

536 sources.json_schema["properties"].pop(name, None) 

537 schema_gaps.setdefault(spec, None) 

538 

539 final = _finalize(best_tree, arb_for_final, survival, sources, model_tag, profile, provenance, fn_name) 

540 if added_features: 

541 final.added_features = added_features # the loop grew the schema by these (review them) 

542 if schema_gaps: 

543 final.schema_gaps = list(schema_gaps) # advisory: prompt to re-open the schema gate 

544 if outcome_gaps: 

545 final.outcome_gaps = list(outcome_gaps) # advisory: prompt to widen the outcome set 

546 _assert_compiles(final) # never hand back a tree that doesn't import 

547 # The ratified examples are the loop's own correctness signal: check the 

548 # converged tree against them and surface any disagreement (§4.1 / §4.5). 

549 if sources.examples: 

550 final.example_report = _check_examples(final, sources.examples) 

551 # Attach the validation set the personas drafted as the loop ran, for a human to 

552 # ratify (§4.5). If the panel happened to propose nothing, fall back to one targeted 

553 # generation call over the gray zones so we never finalize without a proposed set. 

554 if propose_examples: 

555 items = list(proposed.values()) 

556 if not items: 

557 items = [i for i in _propose_examples(backend, sources, final, fn_name) 

558 if _canon(i["input"]) not in existing_inputs] 

559 final.proposed_examples = _stamp_proposed(final, items) 

560 return final

574def _harvest_proposed( 

575 verdicts: list[PersonaVerdict], rnd: int, acc: dict[str, dict], existing_inputs: set[str] 

576) -> None: 

577 """Collect each non-critic persona's drafted test cases into ``acc`` (deduped). 

578 

579 The cases ride along in the verdicts the personas already return, so this is free. 

580 The two counterweights are skipped — they restructure, they don't add cases.""" 

581 for v in verdicts: 

582 if v.persona in (OVERENGINEERING_CRITIC.name, SCHEMA_CRITIC.name, OUTCOME_CRITIC.name): 

583 continue 

584 for ex in v.proposed_tests: 

585 if not ex.input: 

586 continue 

587 key = _canon(ex.input) 

588 if key in existing_inputs or key in acc: 

589 continue 

590 acc[key] = { 

591 "input": ex.input, 

592 "expected": ex.expected, 

593 "rationale": ex.rationale, 

594 "source": f"{v.persona} (round {rnd})", 

595 }

598def _stamp_proposed(tree: DecisionTree, items: list[dict]) -> list[dict]: 

599 """Add the frozen tree's own prediction + the un-ratified status to each case. 

600 

601 The prediction is computed here, deterministically — a machine-authored label never 

602 masquerades as ground truth, and a case whose proposed label differs from the tree's 

603 output is the highest-value one for a human to rule on.""" 

604 from .validate import fn_from_tree 

605 

606 fn = fn_from_tree(tree) 

607 out: list[dict] = [] 

608 for it in items: 

609 try: 

610 prediction = fn(it["input"]) 

611 except Exception as e: # a case that crashes the tree is itself worth surfacing 

612 prediction = f"ERROR: {type(e).__name__}: {e}" 

613 out.append({**it, "tree_prediction": prediction, "status": "proposed"}) 

614 return out

617def _propose_examples( 

618 backend: Backend, sources: Sources, tree: DecisionTree, fn_name: str 

619) -> list[dict]: 

620 """Fallback generator: ask the model for cases targeting the gray zones directly. 

621 

622 Only used when the persona panel happened to propose nothing, so the run never 

623 finalizes without a proposed set. Returns raw {input, expected, rationale} items; 

624 the caller dedups and stamps them.""" 

625 gray = [n.gray_zone for n in tree.nodes if n.gray_zone] 

626 gray_block = "\n".join(f" - {g}" for g in gray) or " (none explicitly flagged)" 

627 user = ( 

628 f"{_sources_block(sources)}\n\n" 

629 f"FROZEN TREE (the decision is now exactly this):\n{tree.to_source()}\n\n" 

630 f"GRAY ZONES still unresolved:\n{gray_block}\n\n" 

631 "Propose a small set of discriminating test cases (at most one or two per gray " 

632 "zone) that pin down these contested cells — the inputs a human reviewer should " 

633 "rule on. Give the outcome you believe is correct for each, as a PROPOSAL to be " 

634 "ratified. Do not duplicate any EXAMPLES already listed." 

635 ) 

636 proposed = backend.complete(EXAMPLE_PROPOSER_SYSTEM, user, ProposedExampleSet) 

637 return [{"input": ex.input, "expected": ex.expected, "rationale": ex.rationale} 

638 for ex in proposed.examples]

641def _score(tree: ProposedTree, items: list[dict], features: list[str], fn_name: str) -> int: 

642 """How many of ``items`` (validation cases) the current tree satisfies (label match). 

643 

644 The proposed cases are the gaps personas found, so this pass-rate rising round over 

645 round is the loop visibly closing them — the per-iteration validation score.""" 

646 if not items: 

647 return 0 

648 from .validate import fn_from_tree, label_match 

649 

650 dt = DecisionTree( 

651 nodes=[DecisionNode(condition=n.condition, outcome=n.outcome) for n in tree.nodes], 

652 default_outcome=tree.default_outcome, features=features, fn_name=fn_name, 

653 ) 

654 fn = fn_from_tree(dt) 

655 passed = 0 

656 for it in items: 

657 try: 

658 pred = fn(it["input"]) 

659 except Exception as e: 

660 pred = f"ERROR: {type(e).__name__}: {e}" 

661 if label_match(pred, it["expected"]): 

662 passed += 1 

663 return passed

667 """Belt-and-suspenders: refuse to return a tree whose source won't import.""" 

668 from .validate import fn_from_tree

670 fn_from_tree(tree) 

671 except SyntaxError as e: 

672 raise RuntimeError(f"refusing to export a tree that does not compile: {e}") from e

675def _select_best( 

676 candidates: list[tuple[ProposedTree, "ProposerArbitration | None", float]], 

677 sources: Sources, prop_items: list[dict], fn_name: str, 

678) -> tuple[ProposedTree, "ProposerArbitration | None"]: 

679 """Pick the candidate tree with the best (ratified rate, proposed rate, persona score). 

680 

681 Re-scores every candidate against the SAME final proposed set, so a round isn't 

682 flattered or penalised by how many cases existed when it ran. Ratified ground truth 

683 is the lexicographic primary — never traded away to match an adversary-proposed label.""" 

684 best = (candidates[0][0], candidates[0][1]) 

685 best_key = (-1.0, -1.0, -1.0) 

686 for tree, arb, mean in candidates: 

687 rat = _agreement(tree, sources, fn_name) 

688 key = ( 

689 rat if rat is not None else 1.0, 

690 _score(tree, prop_items, sources.feature_names, fn_name) / len(prop_items) 

691 if prop_items else 0.0, 

692 mean, 

693 ) 

694 if key > best_key: 

695 best_key = key 

696 best = (tree, arb) 

697 return best

700def _quality(tree: ProposedTree, sources: Sources, prop_items: list[dict], fn_name: str): 

701 """Rank a tree: (ratified pass-rate, proposed pass-rate, fewer nodes). Higher is better. 

702 

703 Ratified ground truth dominates; then how many adversary-proposed cases it passes; then, 

704 at equal correctness, the simpler tree wins — so a node merge that changes no outcome is 

705 an improvement, but a collapse that drops a case is not.""" 

706 rat = _agreement(tree, sources, fn_name) 

707 prop_rate = (_score(tree, prop_items, sources.feature_names, fn_name) / len(prop_items) 

708 if prop_items else 0.0) 

709 return (rat if rat is not None else 1.0, prop_rate, -len(tree.nodes))

712def _agreement(tree: ProposedTree, sources: Sources, fn_name: str) -> float | None: 

713 """Ratified-example agreement for an in-flight ProposedTree (zero LLM — pure Python). 

714 

715 The loop's own correctness signal each round: used both to keep the best tree and 

716 to catch a buzzer-beating regression where a late round breaks a ratified example. 

717 Returns None when there are no examples to score against. 

718 """ 

719 if not sources.examples: 

720 return None 

721 dt = DecisionTree( 

722 nodes=[DecisionNode(condition=n.condition, outcome=n.outcome) for n in tree.nodes], 

723 default_outcome=tree.default_outcome, 

724 features=sources.feature_names, 

725 fn_name=fn_name, 

726 ) 

727 return _check_examples(dt, sources.examples).agreement_rate

730def _finalize( 

731 tree: ProposedTree, 

732 arbitration: ProposerArbitration | None, 

733 survival: dict[str, int], 

734 sources: Sources, 

735 model: str, 

736 profile: str, 

737 provenance: bool, 

738 fn_name: str, 

739) -> DecisionTree: 

740 critic_notes = {} 

741 if arbitration: 

742 for e in arbitration.entries: 

743 if e.persona == OVERENGINEERING_CRITIC.name and e.decision != "rejected": 

744 critic_notes["_global"] = f"{e.decision} — {e.rationale}" 

745 nodes = [ 

746 DecisionNode( 

747 condition=n.condition, 

748 outcome=n.outcome, 

749 rounds_survived=survival.get(_node_key(n.condition), 1), 

750 sources=n.sources, 

751 gray_zone=n.gray_zone, 

752 ) 

753 for n in tree.nodes 

754 ] 

755 if nodes and "_global" in critic_notes: 

756 nodes[0].critic_note = critic_notes["_global"] 

757 return DecisionTree( 

758 nodes=nodes, 

759 default_outcome=tree.default_outcome, 

760 features=sources.feature_names, 

761 fn_name=fn_name, 

762 model=model, 

763 profile=profile, 

764 include_provenance=provenance, 

765 )
