1"""Deterministic tree → .py exporter, callable from the Claude Code skill. 

2 

3The subagent-driven loop (see .claude/skills/temper-skills/SKILL.md) produces a 

4tree as JSON; this turns it into the zero-dependency Python module. No LLM here — 

5this is the deterministic half of the pipeline. 

6 

7 python -m temper_skills.export_tree tree.json route.py 

8 cat tree.json | python -m temper_skills.export_tree - route.py 

9 

10Alongside the module it writes, from ``tree.proposed_examples``: 

11 * ``<stem>.validation.jsonl`` — the committed validation dataset (the debate surface). 

12 Every case, with the tree's own prediction and an ``agrees`` flag. Disagreements are 

13 DATA, not test failures. 

14 * ``test_<stem>.py`` — behavior-lock tests that assert the tree's CURRENT output. Green 

15 by construction; they only go red on a real code regression (drift lock). 

16 * ``test_<stem>_ratified.py`` — assertions of the human-ratified labels ONLY. Emitted 

17 only when at least one case is ``status: "ratified"``. This one CAN go red — and 

18 should, if the tree ever contradicts blessed truth. That is the only sanctioned 

19 test failure; contested/proposed labels never become a failing (or xfail) test. 

20"""

31def _compile(tree: DecisionTree): 

32 # Inline (not validate.fn_from_tree) to keep this module import-light — the 

33 # deterministic export path must not pull in the LLM backends. 

34 ns: dict = {} 

35 exec(compile(tree.to_source(), f"<{tree.fn_name}>", "exec"), ns) 

36 return ns[tree.fn_name]

39def _is_error(pred) -> bool: 

40 return isinstance(pred, str) and pred.startswith("ERROR")

43def enrich_validation(tree: DecisionTree, raw: list[dict]) -> list[dict]: 

44 """Stamp each panel-drafted case with the tree's own prediction, agreement, and status. 

45 

46 The subagent loop authors ``input``/``expected``/``rationale``/``source``; the *prediction*, 

47 the ``agrees`` flag, and the un-ratified default status are computed here deterministically, 

48 so a machine-authored label can never masquerade as ground truth. The result is the 

49 validation dataset — data, not assertions. A disagreement (``agrees is False``) is a number 

50 to review, never a failing test.""" 

51 fn = _compile(tree) 

52 out: list[dict] = [] 

53 for case in raw: 

54 try: 

55 prediction = fn(case["input"]) 

56 except Exception as e: 

57 prediction = f"ERROR: {type(e).__name__}: {e}" 

58 expected = case.get("expected", "") 

59 agrees = None if (expected == "" or _is_error(prediction)) else (expected == prediction) 

60 record = { 

61 "input": case["input"], 

62 "expected": expected, 

63 "rationale": case.get("rationale", ""), 

64 "tree_prediction": prediction, 

65 # None = no proposed label to compare (or tree errored); else does the tree's 

66 # current answer match the proposed/ratified label? 

67 "agrees": agrees, 

68 # "proposed" = panel-authored, ungated; "resolved" = proposer settled a contested 

69 # cell (no domain oracle); "ratified" = a human domain owner blessed the label. 

70 "status": case.get("status", "proposed"), 

71 } 

72 # Pass through loop provenance if the case carries it (round/run id land here in the 

73 # per-round writer; harmless when absent). 

74 for k in ("source", "round", "first_seen_round", "run_id"): 

75 if case.get(k) is not None: 

76 record[k] = case[k] 

77 out.append(record) 

78 return out

85def _canon(inp: dict) -> str: 

86 """Stable dedup key for a case input (feature order-independent).""" 

87 return json.dumps(inp, sort_keys=True, ensure_ascii=False)

90def load_validation(path: str | Path) -> list[dict]: 

91 """Read an existing ``<stem>.validation.jsonl`` (the accumulating dataset), or [] if absent.""" 

92 p = Path(path) 

93 if not p.exists(): 

94 return [] 

95 return [json.loads(ln) for ln in p.read_text().splitlines() if ln.strip()]

98def merge_cases(existing: list[dict], new: list[dict], *, 

99 first_seen_round: int | None = None, run_id: str | None = None) -> list[dict]: 

100 """Merge freshly-proposed cases into the accumulated set, deduped by input. 

101 

102 The first round to find a case owns its label/status and provenance; a later duplicate only 

103 appends its ``source`` (so we can see every persona/round that re-found it). Prediction and 

104 agreement are NOT set here — ``enrich_validation`` recomputes them against the current tree, 

105 so a tree change refreshes every row. Existing rows keep their order; new rows append.""" 

106 by_key: dict[str, dict] = {} 

107 order: list[str] = [] 

108 for row in existing: 

109 k = _canon(row["input"]) 

110 if k not in by_key: 

111 by_key[k] = dict(row) 

112 order.append(k) 

113 for case in new: 

114 k = _canon(case["input"]) 

115 if k in by_key: 

116 src = case.get("source") 

117 if src: 

118 row = by_key[k] 

119 srcs = [s for s in (row.get("source", "").split(";")) if s] 

120 if src not in srcs: 

121 srcs.append(src) 

122 row["source"] = ";".join(srcs) 

123 continue 

124 row = { 

125 "input": case["input"], 

126 "expected": case.get("expected", ""), 

127 "rationale": case.get("rationale", ""), 

128 "status": case.get("status", "proposed"), 

129 } 

130 if case.get("source"): 

131 row["source"] = case["source"] 

132 if first_seen_round is not None: 

133 row["first_seen_round"] = first_seen_round 

134 if run_id is not None: 

135 row["run_id"] = run_id 

136 by_key[k] = row 

137 order.append(k) 

138 return [by_key[k] for k in order]

141def _dataset_score(enriched: list[dict]) -> dict: 

142 comparable = [c for c in enriched if c["agrees"] is not None] 

143 agree = sum(1 for c in comparable if c["agrees"]) 

144 return { 

145 "total": len(enriched), 

146 "comparable": len(comparable), 

147 "agree": agree, 

148 "disputes": len(comparable) - agree, 

149 "ratified": sum(1 for c in enriched if c.get("status") == "ratified"), 

150 }

153def write_dataset_and_tests(tree: DecisionTree, out_py: str, enriched: list[dict]) -> dict: 

154 """Write ``<stem>.validation.jsonl`` + the behavior-lock test + (only if any case is 

155 ratified) the ratified-truth test. Shared by ``export_tree`` (final) and 

156 ``update_validation`` (per round) so both produce byte-identical artifacts. Returns a 

157 score summary. A stale ``test_<stem>_ratified.py`` is removed when nothing is ratified.""" 

158 out_path = Path(out_py) 

159 stem = str(out_path.with_suffix("")) 

160 Path(stem + ".validation.jsonl").write_text( 

161 "".join(json.dumps(r, ensure_ascii=False) + "\n" for r in enriched)) 

162 

163 Path(out_path.with_name("test_" + out_path.name)).write_text( 

164 render_behavior_lock(out_path.stem, tree.fn_name, enriched)) 

165 

166 rat_path = out_path.with_name("test_" + out_path.stem + "_ratified.py") 

167 rat_src = render_ratified(out_path.stem, tree.fn_name, enriched) 

168 if rat_src is not None: 

169 rat_path.write_text(rat_src) 

170 elif rat_path.exists(): 

171 rat_path.unlink() # nothing ratified any more — don't leave a stale test 

172 

173 return _dataset_score(enriched)

176def _note(c: dict) -> str: 

177 return (c.get("source", "") + (" — " + c["rationale"] if c.get("rationale") else "")).strip(" —")

180def render_behavior_lock(module: str, fn_name: str, enriched: list[dict]) -> str: 

181 """Committed pytest that locks the tree's CURRENT output for every case. 

182 

183 Green by construction: it asserts what the tree returns, so it can only go red on a real 

184 code regression (a drift lock). Debates never live here — a contested case is still locked 

185 to the tree's own answer; the disagreement with the proposed label is recorded in the 

186 validation dataset, not as a test.""" 

187 rows = [c for c in enriched if not _is_error(c["tree_prediction"])] 

188 L = [ 

189 "# Auto-generated by temper-skills — DO NOT EDIT. Regenerate with export_tree.", 

190 f"# Behavior lock for {fn_name}: each case asserts the tree's CURRENT output.", 

191 "# Always green by construction — a failure here means the tree drifted (a real regression).", 

192 "# Debates are NOT here; they live in the .validation.jsonl dataset.", 

193 "import pytest", 

194 f"from {module} import {fn_name}", 

195 "", 

196 "# (input, tree_output, status, note)", 

197 "LOCKED = [", 

198 ] 

199 for c in rows: 

200 L.append(f" ({c['input']!r}, {c['tree_prediction']!r}, {c['status']!r}, {_note(c)!r}),") 

201 L += [ 

202 "]", 

203 "", 

204 '@pytest.mark.parametrize("case,expected,status,note", LOCKED, ' 

205 'ids=[c[3][:60] for c in LOCKED])', 

206 f"def test_{fn_name}_behavior(case, expected, status, note):", 

207 f" assert {fn_name}(case) == expected", 

208 "", 

209 ] 

210 return "\n".join(L)

213def render_ratified(module: str, fn_name: str, enriched: list[dict]) -> str | None: 

214 """Committed pytest asserting the tree honors every RATIFIED validation label. 

215 

216 Unlike the behavior lock, this CAN go red — and should, if the tree ever contradicts a 

217 human-blessed answer. That is the only sanctioned test failure in the pipeline. Returns 

218 None when nothing is ratified yet (no file is written), so a run never carries an empty or 

219 xfail placeholder — open disputes stay as data in the dataset until a human rules.""" 

220 rows = [c for c in enriched 

221 if c.get("status") == "ratified" and c.get("expected") not in ("", None) 

222 and not _is_error(c["tree_prediction"])] 

223 if not rows: 

224 return None 

225 L = [ 

226 "# Auto-generated by temper-skills — DO NOT EDIT. Regenerate with export_tree.", 

227 f"# Ratified-truth tests for {fn_name}: assert the tree matches HUMAN-blessed labels.", 

228 "# This file CAN fail — a red test here means the tree contradicts ratified ground truth.", 

229 "# Comparison is leading-label (temper-skills' label_match): a descriptive outcome like", 

230 "# 'no — toxic' honors the ratified label 'no'.", 

231 "import pytest", 

232 f"from {module} import {fn_name}", 

233 "", 

234 "def _label(s):", 

235 " head = str(s).strip().split()[0] if str(s).strip() else ''", 

236 " return head.lower().rstrip(':,.—-')", 

237 "", 

238 "# (input, ratified_label, note)", 

239 "RATIFIED = [", 

240 ] 

241 for c in rows: 

242 L.append(f" ({c['input']!r}, {c['expected']!r}, {_note(c)!r}),") 

243 L += [ 

244 "]", 

245 "", 

246 '@pytest.mark.parametrize("case,expected,note", RATIFIED, ids=[c[2][:60] for c in RATIFIED])', 

247 f"def test_{fn_name}_ratified(case, expected, note):", 

248 f" assert _label({fn_name}(case)) == _label(expected)", 

249 "", 

250 ] 

251 return "\n".join(L)

254def tree_from_dict(data: dict) -> DecisionTree: 

255 nodes = [ 

256 DecisionNode( 

257 condition=n["condition"], 

258 outcome=n["outcome"], 

259 rounds_survived=int(n.get("rounds_survived", 1)), 

260 sources=list(n.get("sources", [])), 

261 critic_note=n.get("critic_note"), 

262 gray_zone=n.get("gray_zone"), 

263 ) 

264 for n in data["nodes"] 

265 ] 

266 return DecisionTree( 

267 nodes=nodes, 

268 default_outcome=data["default_outcome"], 

269 features=list(data.get("features", [])), 

270 fn_name=data.get("fn_name", "decide"), 

271 model=data.get("model", "claude-code-subagents"), 

272 profile=data.get("profile", "standard"), 

273 constraints_version=data.get("constraints_version", "v1.0"), 

274 include_provenance=bool(data.get("include_provenance", True)), 

275 )

278def main(argv: list[str] | None = None) -> int: 

279 argv = argv if argv is not None else sys.argv[1:] 

280 if len(argv) != 2: 

281 print("usage: python -m temper_skills.export_tree <tree.json|-> <out.py>", file=sys.stderr) 

282 return 2 

283 src, out = argv 

284 raw = sys.stdin.read() if src == "-" else open(src).read() 

285 data = json.loads(raw) 

286 tree = tree_from_dict(data) 

287 tree.export(out) 

288 print(f"exported {out} ({len(tree.nodes)} nodes)") 

289 

290 # Case source: fold tree.json's proposed_examples into any dataset the per-round loop already 

291 # accumulated on disk, so a final export refreshes predictions without clobbering provenance. 

292 stem = str(Path(out).with_suffix("")) 

293 existing = load_validation(stem + ".validation.jsonl") 

294 from_tree = data.get("proposed_examples") or [] 

295 if existing or from_tree: 

296 merged = merge_cases(existing, from_tree) 

297 enriched = enrich_validation(tree, merged) 

298 s = write_dataset_and_tests(tree, out, enriched) 

299 score = f"{s['agree']}/{s['comparable']}" if s["comparable"] else "n/a" 

300 print(f"wrote {s['total']} validation case(s) → {stem}.validation.jsonl") 

301 print(f" tree agrees with {score} labelled case(s); " 

302 f"{s['disputes']} open disagreement(s) (data, not failures — review to ratify)") 

303 print(f"wrote behavior-lock tests (always green) → " 

304 f"{Path(out).with_name('test_' + Path(out).name)}") 

305 if s["ratified"]: 

306 print(f"wrote ratified-truth tests ({s['ratified']} case(s), can fail on regression)") 

307 return 0
