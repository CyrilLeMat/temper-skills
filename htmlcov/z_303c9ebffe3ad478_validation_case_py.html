1"""The validation-case contract — the one place the dataset row shape is defined. 

2 

3A case is born in the loop (a persona proposes input + expected + rationale), merged 

4into the accumulating ``<stem>.validation.jsonl`` (dedup by input, provenance appended), 

5and enriched against the current tree (prediction + agreement recomputed every run). 

6Those three stages live in distill.py, export_tree.py, and skill_render.py — before this 

7module each re-declared the keys by hand, so adding a field meant three coordinated 

8edits and a rename was a silent data drop. 

9 

10Rows stay plain dicts on disk and across module boundaries (JSONL); construct them via 

11``ValidationCase(...).to_record()`` and parse via ``from_dict`` so the shape is checked 

12in one place. Stdlib-only ON PURPOSE: export_tree/skill_render/update_validation are 

13vendored into the subagent skill's ``scripts/`` (no third-party deps), so this must be a 

14dataclass, not Pydantic. 

15""" 

16 

17from __future__ import annotations 

18 

19import json 

20from dataclasses import dataclass, fields 

21 

22# proposed = panel-authored, ungated; never a CI gate. 

23# resolved = the proposer settled a contested cell (no domain oracle available). 

24# ratified = a human domain owner blessed the label — authoritative ground truth. 

25STATUSES = ("proposed", "resolved", "ratified") 

26 

27# Loop provenance, carried through merges untouched (in this record order). 

28PROVENANCE_KEYS = ("source", "round", "first_seen_round", "run_id") 

29 

30 

31def canon(inp: dict) -> str: 

32 """Stable dedup key for a case input (feature order-independent).""" 

33 return json.dumps(inp, sort_keys=True, ensure_ascii=False, default=str) 

34 

35 

36@dataclass 

37class ValidationCase: 

38 input: dict 

39 expected: str = "" 

40 rationale: str = "" 

41 status: str = "proposed" 

42 # Provenance (optional; see PROVENANCE_KEYS): 

43 source: str | None = None 

44 round: int | None = None 

45 first_seen_round: int | None = None 

46 run_id: str | None = None 

47 # Enrichment — computed against the CURRENT tree by enrich_validation, never 

48 # authored. agrees is tri-state: None = no label to compare (or the tree errored). 

49 tree_prediction: str | None = None 

50 agrees: bool | None = None 

51 

52 @classmethod 

53 def from_dict(cls, row: dict) -> ValidationCase: 

54 """Parse a loose row, tolerating unknown keys — old datasets stay loadable.""" 

55 known = {f.name for f in fields(cls)} 

56 return cls(**{k: v for k, v in row.items() if k in known}) 

57 

58 def to_record(self) -> dict: 

59 """The canonical dict row. Key order is stable (it is the on-disk JSONL diff 

60 surface); enrichment keys appear only on enriched rows (tree_prediction set), 

61 provenance keys only when present.""" 

62 rec: dict = {"input": self.input, "expected": self.expected, 

63 "rationale": self.rationale} 

64 if self.tree_prediction is not None: 

65 rec["tree_prediction"] = self.tree_prediction 

66 rec["agrees"] = self.agrees 

67 rec["status"] = self.status 

68 if self.source: 

69 rec["source"] = self.source 

70 for key in ("round", "first_seen_round", "run_id"): 

71 val = getattr(self, key) 

72 if val is not None: 

73 rec[key] = val 

74 return rec 

75 

76 def key(self) -> str: 

77 return canon(self.input)
